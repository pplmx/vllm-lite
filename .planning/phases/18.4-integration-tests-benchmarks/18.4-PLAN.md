//! Multi-Model Speculative Decoding (v18.0) — integration test harness
//!
//! Composes DraftModelRegistry + DraftResolver + MemoryBudget into a single
//! testable surface. Uses stub backends that record calls and support
//! deterministic outputs and configurable failure modes.

use std::sync::{Arc, Mutex};
use std::sync::atomic::{AtomicU32, AtomicU64, Ordering};
use std::collections::HashMap;
use vllm_core::speculative::{
    DraftId, DraftLoader, DraftModelRegistry, DraftRegistryError, DraftSpec,
    MemoryBudget, MemoryBudgetExceeded, ResolvedDraft, DraftResolver,
};
use vllm_core::metrics::EnhancedMetricsCollector;
use vllm_traits::{BatchOutput, ModelBackend, Result as ModelResult, ...};

/// Configurable stub backend. Tracks forward() call count + last input,
/// returns deterministic outputs based on `id`, supports per-instance
/// "fail next N calls" for runtime-error testing.
pub struct StubBackend {
    pub id: String,
    pub fail_next_n: AtomicU32,
    pub call_count: AtomicU64,
}

impl ModelBackend for StubBackend {
    fn forward(&mut self, ...) -> ModelResult<BatchOutput> {
        self.call_count.fetch_add(1, Ordering::Relaxed);
        let n = self.fail_next_n.load(Ordering::Relaxed);
        if n > 0 {
            self.fail_next_n.fetch_sub(1, Ordering::Relaxed);
            return Err(ModelError::new("simulated draft runtime error"));
        }
        Ok(/* deterministic output */)
    }
    // ... other trait methods return simple values
}

/// Stub loader that returns the requested backend from a map, or
/// `LoadFailed` if the id isn't in the map.
pub struct MapLoader {
    pub backends: Mutex<HashMap<DraftId, Box<dyn ModelBackend>>>,
}

impl DraftLoader for MapLoader {
    fn load(&self, id: &DraftId) -> Result<Box<dyn ModelBackend>, DraftRegistryError> {
        self.backends
            .lock()
            .unwrap()
            .remove(id)
            .ok_or_else(|| DraftRegistryError::LoadFailed(format!("no stub for {id}")))
    }
}

/// Builder for a complete test setup.
pub struct Harness {
    pub registry: Arc<DraftModelRegistry>,
    pub budget: Arc<MemoryBudget>,
    pub resolver: Arc<DraftResolver>,
    pub metrics: Arc<EnhancedMetricsCollector>,
    pub loader: Arc<MapLoader>,
    pub self_spec: Arc<Mutex<Box<dyn ModelBackend>>>,
}

pub fn harness_with_budget(total_bytes: u64) -> Harness { ... }
pub fn harness_unlimited() -> Harness { ... }

### 3. Benchmark

**`crates/core/benches/multi_draft_speculative.rs`** (new file):

```rust
//! Multi-Model Speculative Decoding (v18.0) benchmark.
//!
//! Compares three configurations:
//! 1. No draft (baseline pure target decode)
//! 2. Self-spec draft (v17 baseline)
//! 3. External draft (v18.0)

use criterion::{criterion_group, criterion_main, Criterion, BenchmarkId};

fn bench_throughput(c: &mut Criterion) {
    let mut group = c.benchmark_group("multi_draft_throughput");

    for config in ["no_draft", "self_spec", "external_draft"] {
        group.bench_with_input(
            BenchmarkId::from_parameter(config),
            config,
            |b, &cfg| {
                let harness = setup_for_config(cfg);
                // Warmup
                for _ in 0..5 { run_iteration(&harness); }
                b.iter(|| run_iteration(&harness));
            },
        );
    }

    group.finish();
}

fn run_iteration(harness: &Harness) {
    // Simulate 16-step decode with the configured draft path
    for _ in 0..16 {
        harness.step();
    }
}

criterion_group!(benches, bench_throughput);
criterion_main!(benches);
```
