```rust
let harness = TestHarness::new()
    .kv_blocks(128)
    .max_batch_size(8)
    .build();

let seq_id = harness.add_test_request(vec![1, 2, 3], 10);
```

### Mock Models (TEST-02)
- `NeverProgressModel` - Returns same token, blocks indefinitely (timeout testing)
- `SlowModel` - Configurable delay for backpressure testing

```rust
use vllm_testing::{NeverProgressModel, SlowModel};
use std::time::Duration;

// Blocks indefinitely
let never = NeverProgressModel::new(1);

// 10ms delay per forward
let slow = SlowModel::new(Duration::from_millis(10));
```

### RequestFactory (TEST-03)
```rust
use vllm_testing::RequestFactory;

let mut factory = RequestFactory::new()
    .min_tokens(64)
    .max_tokens(512)
    .max_max_tokens(32);

// Create single request
let request = factory.create();

// Create batch
let requests = factory.create_batch(10);
```
