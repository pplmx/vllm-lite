```rust
#[derive(Clone, Debug)]
pub struct SpeculationConfig {
    pub draft_count: usize,
    pub max_depth: usize,
    pub temperature: f32,

    pub draft_layers: Option<usize>,

    fn default() -> Self {
        Self {
            draft_count: 4,
            max_depth: 8,
            temperature: 0.0,
            target_model: String::new(),
            draft_layers: None,
        }
    }
}

impl SpeculationConfig {
    pub fn builder() -> SpeculationConfigBuilder {
        SpeculationConfigBuilder::default()

#[derive(Default)]

    config: SpeculationConfig,

    pub fn draft_count(mut self, count: usize) -> Self { ... }

    pub fn temperature(mut self, temp: f32) -> Self { ... }
```

## Task 3: Create RejectionStrategy Enum

**Requirements:** SPEC-01.4

**File:** `crates/core/src/speculative/strategy.rs`

**Implementation:**
```rust
#[derive(Clone, Debug, PartialEq)]
pub enum RejectionStrategy {
    TokenLevel,
    BlockLevel { block_size: usize },
}

impl RejectionStrategy {
    pub fn should_accept(&self, draft_prob: f32, target_prob: f32) -> bool {
        match self {
            RejectionStrategy::TokenLevel => target_prob >= draft_prob,
            RejectionStrategy::BlockLevel { .. } => target_prob > draft_prob,
        }
    }
}

impl Default for RejectionStrategy {
    fn default() -> Self {
        RejectionStrategy::TokenLevel
    }
}
```

**Tests:**
- Unit test: TokenLevel acceptance logic
- Unit test: BlockLevel acceptance logic

```rust
use vllm_traits::ModelBackend;
use super::{DraftVerifier, SpeculationConfig, RejectionStrategy};
use crate::types::Batch;

pub struct SpeculativeModel<M: ModelBackend> {
    target_model: M,
    verifier: Box<dyn DraftVerifier>,
    config: SpeculationConfig,
    strategy: RejectionStrategy,
}

impl<M: ModelBackend> SpeculativeModel<M> {
    pub fn new(
        target_model: M,
        verifier: Box<dyn DraftVerifier>,
        config: SpeculationConfig,
        strategy: RejectionStrategy,
    ) -> Self {
        Self {
            target_model,
            verifier,
            config,
            strategy,
        }
    }

    pub fn config(&self) -> &SpeculationConfig {
        &self.config
    }

    pub fn set_strategy(&mut self, strategy: RejectionStrategy) {
        self.strategy = strategy;
    }
}
```
